	\maketitle

	Дан логарифм функции правдоподобия:
	\[
	\ell(a, b) = 3\ln a + \ln b - 2a - 2b - 8.
	\]
	
	\begin{enumerate}[label = \alph*)]
		\item Найдите $\hat{a}_{ML}$, $\hat{b}_{ML}$ и покажите, что в найденной точке действительно достигается максимум правдоподобия.
		\item Постройте $95\%$-ый доверительный интервал для $b$.
		\item Проверьте гипотезу
		\[
		H_0: \begin{cases}
			\begin{pmatrix}
				a \\
				b
			\end{pmatrix} = 
			\begin{pmatrix}
				1 \\
				2
			\end{pmatrix}
		\end{cases}
		\]
		против 
		\[
		H_0: \begin{cases}
			\begin{pmatrix}
				a \\
				b
			\end{pmatrix} \ne 
			\begin{pmatrix}
				1 \\
				2
			\end{pmatrix}
		\end{cases}
		\]
		на уровне значимости 5\% (1) при помощи теста LR и (2) теста LM или W на ваш выбор.
		\item Проверьте гипотезу
		\[
		H_0: a = 2b
		\]
		против
		\[
		H_1: a \ne 2b
		\]
		на уровне значимости 5\% при помощи теста LR.
	\end{enumerate}

	Критические значения: $Z_{0.975} = 1.96$, $\chi^2_1 = 3.84$, $\chi^2_2 = 5.99$, $\chi^2_3 = 7.81$.
